671  inline bool Operation::increment(std::vector<int>& x, int position, std::vector<int>& min, std::vector<int>& max) {
672  if (x.empty() or position < 0) return true;
673 
674  bool overflow = false;
675  if (x.at(position) < max.at(position)) {
676  ++x.at(position);
677  } else {
678  overflow = (position == 0);
679  x.at(position) = min.at(position);
680  --position;
681  if (position >= 0) {
682  overflow = increment(x, position, min, max);
683  }
684  }
685  return overflow;

754  inline std::vector<std::string> Operation::split(const std::string& str,
755  const std::string& delimiter, bool ignoreEmpty) {
756  std::vector<std::string> result;
757  if (str.empty() or delimiter.empty()) {
758  result.push_back(str);
759  return result;
760  }
761  std::string::const_iterator position = str.begin(), next = str.begin();
762  while (next != str.end()) {
763  next = std::search(position, str.end(), delimiter.begin(), delimiter.end());
764  std::string token(position, next);
765  if (not (token.empty() and ignoreEmpty)) {
766  result.push_back(token);
767  }
768  if (next != str.end()) {
769  position = next + delimiter.size();
770  }
771  }
772  return result;
773  }

775  inline std::string Operation::trim(const std::string& text) {
776  if (text.empty()) return text;
777  if (not (std::isspace(text.at(0)) or std::isspace(text.at(text.size() - 1))))
778  return text;
779  std::size_t start = 0, end = text.size() - 1;
780  while (start <= end and std::isspace(text.at(start))) {
781  ++start;
782  }
783  while (end >= start and std::isspace(text.at(end))) {
784  --end;
785  }
786  std::size_t length = end - start + 1;
787  if (length <= 0) return "";
788  return text.substr(start, length);
789  }
