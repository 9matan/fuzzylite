260  T x1, T y1, ...) {
261  std::vector<scalar> xy(argc);
262  xy.at(0) = x1;
263  xy.at(1) = y1;
264  va_list args;
265  va_start(args, y1);
266  for (int i = 2; i < argc; ++i) {
267  xy.at(i) = (scalar) va_arg(args, T);
268  }
269  va_end(args);
270 
271  FL_unique_ptr<Discrete> result(new Discrete(name));
272  if (xy.size() % 2 != 0) {
273  result->setHeight(xy.back());
274  xy.pop_back();
275  }
276  result->setXY(toPairs(xy));
277  return result.release();
